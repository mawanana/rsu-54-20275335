CREATE TABLE `cricket_info`.matches (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    team_1 VARCHAR(255),
    team_2 VARCHAR(255),
    winner VARCHAR(255),
    margin VARCHAR(255),
    ground VARCHAR(255),
    match_date VARCHAR(255),
    match_id VARCHAR(255),
	url VARCHAR(255),
    team_1_score INT,
    team_1_wicket INT,
    team_1_over VARCHAR(255),
    team_2_score INT,
    team_2_wicket INT,
    team_2_over VARCHAR(255),
	team_1_total_over VARCHAR(255),
    team_2_total_over VARCHAR(255),
    first_bat_team VARCHAR(255),
    second_bat_team VARCHAR(255),
    harvest_status VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `cricket_info`.batting (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    match_id VARCHAR(255),
    team VARCHAR(255),
    opposite_team VARCHAR(255),
    player VARCHAR(255),
    profile_url VARCHAR(255),
    batting_position VARCHAR(255),
    captain VARCHAR(255),
    wicket_keeper VARCHAR(255),
    dismissal_method VARCHAR(255),
    bowler VARCHAR(255),
    dismissal_participate_player VARCHAR(255),
    runs VARCHAR(255),
    balls VARCHAR(255),
    minutes VARCHAR(255),
    fours VARCHAR(255),
    sixes VARCHAR(255),
    strike_rate VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `cricket_info`.bowling (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    match_id VARCHAR(255),
    team VARCHAR(255),
    opposite_team VARCHAR(255),
    player VARCHAR(255),
    profile_url VARCHAR(255),
    overs VARCHAR(255),
    maidens VARCHAR(255),
    runs VARCHAR(255),
    wickets VARCHAR(255),
    economy_rate VARCHAR(255),
    dot VARCHAR(255),
    fours VARCHAR(255),
    sixes VARCHAR(255),
    wide_balls VARCHAR(255),
    no_balls VARCHAR(255),
    bowling_position VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `cricket_info`.bowling_wickets (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    match_id VARCHAR(255),
    team VARCHAR(255),
    opposite_team VARCHAR(255),
    player VARCHAR(255),
    profile_url VARCHAR(255),
    bowling_position VARCHAR(255),
    overs VARCHAR(255),
    out_player VARCHAR(255),
    runs VARCHAR(255),
    wicket_position VARCHAR(255),
    total_wickets VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `cricket_info`.ball_by_ball (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    match_id VARCHAR(255),
    team VARCHAR(255),
    opposite_team VARCHAR(255),
    ball VARCHAR(255),
    runs VARCHAR(255),
    other_info VARCHAR(255),
    bowling_player VARCHAR(255),
    batting_player VARCHAR(255),
    shorts VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `cricket_info`.master (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    `match_id` VARCHAR(255),
    `date` DATE,
    `city` VARCHAR(255),
    `winner` VARCHAR(255),
    `winner_method` VARCHAR(255),
    `winner_margin` INT,
    `total_overs` INT,
    `player_of_match` VARCHAR(255),
    `team1` VARCHAR(255),
    `team2` VARCHAR(255),
    `team1_players` VARCHAR(255),
    `team2_players` VARCHAR(255),
    `first_bat` VARCHAR(255),
    `first_ball` VARCHAR(255),
    `venue` VARCHAR(255),
    `inning` VARCHAR(255),
    `over` INT,
    `deliveries` INT,
    `batter` VARCHAR(255),
    `bowler` VARCHAR(255),
    `runs_batter` INT,
    `extras` INT,
    `total_runs` INT,
    `wicket_player_out` VARCHAR(255),
    `wicket_kind` VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
